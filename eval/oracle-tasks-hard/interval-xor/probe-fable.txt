def symmetric_difference_length(A, B):
    def total(lst):
        return sum(e - s + 1 for s, e in lst)

    # Intersection size via two-pointer sweep over integer intervals
    i = j = 0
    inter = 0
    while i < len(A) and j < len(B):
        s = max(A[i][0], B[j][0])
        e = min(A[i][1], B[j][1])
        if s <= e:
            inter += e - s + 1
        if A[i][1] < B[j][1]:
            i += 1
        else:
            j += 1

    return total(A) + total(B) - 2 * inter
